cmake_minimum_required(VERSION 3.28)
project(Dedekind VERSION 1.0.0 LANGUAGES CXX)

# --- Instrumentation and Warnings ---
# Hard-coded for Clang 21 as used in the CI pipeline
set(DEDEKIND_WARNING_FLAGS
    -Wall -Wextra -Wpedantic -Wshadow -Wunused 
    -Wunreachable-code -Wconversion -Werror
    -fprofile-instr-generate 
    -fcoverage-mapping
)

# Ensure the linker also uses the profiling runtime
add_link_options(-fprofile-instr-generate)
# Force C++23 standard for Clang 21
set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
# For clangd:
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
set(CMAKE_CXX_EXTENSIONS OFF)
# Disable global scanning to protect FetchContent/dependencies
set(CMAKE_CXX_SCAN_FOR_MODULES OFF)

# Enable testing with Catch2:

  GIT_TAG        v3.4.0
)

FetchContent_MakeAvailable(Catch2)

# CRITICAL: Use the variable name generated by FetchContent (lowercase)
list(APPEND CMAKE_MODULE_PATH "${catch2_SOURCE_DIR}/extras")

# This provides the catch_discover_tests command
include(Catch)

enable_testing()


# Run build on a per module basis. I.e.
# Module A -> Test A -> Module B -> Test B -> ...
# This macro describes the build of one module.
function(add_dedekind_layer LAYER_NAME DEPENDS_ON)
    set(LIB_TARGET dedekind_${LAYER_NAME})
    set(TEST_TARGET test_${LAYER_NAME})

    file(GLOB MODULE_SOURCES CONFIGURE_DEPENDS "src/main/modules/dedekind/${LAYER_NAME}/*.cppm")
    file(GLOB TEST_SOURCES   CONFIGURE_DEPENDS "src/test/cpp/modules/dedekind/${LAYER_NAME}/*_test.cpp")

    # Use STATIC but force the linker to keep all instrumented code
    add_library(${LIB_TARGET} STATIC)
    set_target_properties(${LIB_TARGET} PROPERTIES CXX_SCAN_FOR_MODULES ON)
    
    # This is the "magic" line for coverage and module initializers
    set_target_properties(${LIB_TARGET} PROPERTIES INTERFACE_LINK_OPTIONS "-Wl,--whole-archive")

    target_sources(${LIB_TARGET}
        PUBLIC 
        FILE_SET CXX_MODULES 
        BASE_DIRS src/main/modules
        FILES ${MODULE_SOURCES}
    )

    target_compile_options(${LIB_TARGET} PRIVATE ${DEDEKIND_WARNING_FLAGS})
    
    if(DEPENDS_ON)
        target_link_libraries(${LIB_TARGET} PUBLIC ${DEPENDS_ON})
    endif()

    if(NOT TEST_SOURCES)
        message(FATAL_ERROR "No tests found for layer: ${LAYER_NAME}")
    endif()

    add_executable(${TEST_TARGET} ${TEST_SOURCES})
    set_target_properties(${TEST_TARGET} PROPERTIES CXX_SCAN_FOR_MODULES ON)
    
    # Linking here now pulls in the logic and all dependencies correctly
    target_link_libraries(${TEST_TARGET} PRIVATE ${LIB_TARGET} Catch2::Catch2WithMain)

    add_test(NAME ${TEST_TARGET} COMMAND ${TEST_TARGET})
    
    add_custom_command(TARGET ${TEST_TARGET} POST_BUILD
        COMMAND $<TARGET_FILE:${TEST_TARGET}>
        COMMENT "Enforcing ${LAYER_NAME} ontology..."
        USES_TERMINAL
    )
endfunction()

# --- THE PROPOSITIONS (Serialized by Libraries) ---
add_dedekind_layer(category "")
add_dedekind_layer(ontology dedekind_category)
add_dedekind_layer(sets dedekind_ontology)


# --- 3. THE FINAL ASSEMBLY (The Public Interface) ---

# The book of math according to the dedekind project.
# Each chapter corresponds the book to an object file which in turn
# corresponds to a module interface unit.
# The "dedekind" library is the final product
add_library(dedekind INTERFACE)
add_dependencies(dedekind dedekind_category dedekind_ontology dedekind_sets)
# No source files needed here; it just bundles the objects
target_link_libraries(dedekind INTERFACE dedekind_ontology dedekind_sets)
target_compile_features(dedekind INTERFACE cxx_std_23)

target_compile_options(dedekind INTERFACE ${DEDEKIND_WARNING_FLAGS})

install(TARGETS dedekind dedekind_category dedekind_ontology dedekind_sets
    EXPORT dedekind-targets
    # Ensure CXX_MODULES are installed for all targets
    FILE_SET CXX_MODULES DESTINATION lib/modules/dedekind
)

install(EXPORT dedekind-targets
    FILE dedekindConfig.cmake
    NAMESPACE dedekind::
    DESTINATION lib/cmake/dedekind
)

# --- Doxygen Documentation (Maven-style) ---
find_package(Doxygen)
if (DOXYGEN_FOUND)
    # Define where Doxygen looks for code and where it outputs HTML
    set(DOXYGEN_INPUT_DIR ${CMAKE_CURRENT_SOURCE_DIR}/src/main/modules)
    set(DOXYGEN_OUTPUT_DIR ${CMAKE_CURRENT_BINARY_DIR}/docs)

    # Customise for C++20 Modules (requires manual Doxyfile settings if needed)
    set(DOXYGEN_EXTENSION_MAPPING "cppm=C++") # Treat .cppm files as C++
    set(DOXYGEN_FILE_PATTERNS "*.cppm" "*.cpp")

    # Hide symbols not marked with 'export' or documented specifically
    set(DOXYGEN_EXTRACT_ALL NO)
    set(DOXYGEN_EXTRACT_PRIVATE NO)
    set(DOXYGEN_EXTRACT_STATIC NO)

    # Improves how Doxygen resolves imports/includes for modules
    set(DOXYGEN_BUILTIN_STL_SUPPORT YES)
    set(DOXYGEN_CPP_CLI_SUPPORT NO)
    set(DOXYGEN_SIP_SUPPORT NO)

    # Crucial for ETCS/Topos documentation: 
    # Allows Doxygen to parse LaTeX-style math if you use it in the future
    set(DOXYGEN_USE_MATHJAX YES) 

    # Ensure the "Level" sections show up in the hierarchy
    set(DOXYGEN_TOC_INCLUDE_HEADINGS 3)

    # Automatically pulls "1.0.0" from the project() call above
    set(DOXYGEN_PROJECT_NUMBER ${PROJECT_VERSION})
    
    # Optional: Add a project brief (like a Maven description)
    set(DOXYGEN_PROJECT_BRIEF "A structuralist C++23 Module Library")
    
    # Use the FindDoxygen built-in macro to configure a Doxyfile automatically
    doxygen_add_docs(docs
        ${DOXYGEN_INPUT_DIR}
        COMMENT "Generating API documentation with Doxygen"
    )
    
endif()
